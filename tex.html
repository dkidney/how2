manually install texlive

  • http://tug.org/texlive/doc/install-tl.html
  • https://riptutorial.com/latex/example/7452/installation-and-setup
# download the installer and cd to the downloaded directory
cd ~/Downloads/
wget "http://mirror.ctan.org/systems/texlive/tlnet/install-tl-unx.tar.gz"
tar -xf install-tl-unx.tar.gz
cd ~/Downloads/
#ls . | grep "install-tl-2"
#cd install-tl-20190731
cd install-tl-*/

# manually set the installation path if you want to 
TEXLIVE_INSTALL_PREFIX=/usr/local/texlive

# start the installer
sudo ./install-tl -gui text

# press I when promoted to complete the install OR
# press P to save the texlive.profile to the currect directory
# which you can then edit if you wish and then use to configure the installation
sudo ./install-tl -gui text --profile texlive.profile

TEXMFHOME: directory for user-specific files

tlmgr

  • command line package manager
brew cask install tex-live-utility # a gui for tlmgr
# update tlmgr itself
tlmgr update --self # might need sudo

# see a list of the updatable packages
tlmgr update --list

# update all packages
tlmgr update --all # might need sudo

# install a package
tlmgr install ctex # might need sudo
