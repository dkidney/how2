inv_sqrt_mat <- function(x) {
  E <- eigen(x)
  if (!all(E$values > 0)) {
    stop("Matric must be positive semi-definite")
  }
  x_sqrt <- E$vectors %*% diag(sqrt(E$values)) %*% t(E$vectors)
  solve(x_sqrt)
}
