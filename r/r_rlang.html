evaluation

expr(s) / enexpr(s)

expr(s)

  • capture expressions that you supply
  • quote contents
z <- expr(x + y)
z
class(z)
#  x + y
#  [1] "call"
z <- exprs(x + y, a + b)
z
class(z)
#  [[1]]
#  x + y
#  
#  [[2]]
#  a + b
#  
#  [1] "list"

enexpr(s)

  • capture expressions that your user supplied
  • call from within a function
  • quote what the user passed to an argument
f <- function(x) {
  enexpr(x)
}
z <- f(x + y)
z
class(z)
#  x + y
#  [1] "call"
f <- function(x, y) {
  enexprs(x, y)
}
z <- f(x + y, a + b)
z
class(z)
#  [[1]]
#  x + y
#  
#  [[2]]
#  a + b
#  
#  [1] "list"

quo(s) / enquo(s)

quo(s)

  • quote contents as a quosure
z <- quo(x + y)
z
class(z)
#  <quosure>
#  expr: ^x + y
#  env:  global
#  [1] "quosure" "formula"
z <- quos(x + y, a + b)
z
class(z)
#  <list_of<quosure>>
#  
#  [[1]]
#  <quosure>
#  expr: ^x + y
#  env:  global
#  
#  [[2]]
#  <quosure>
#  expr: ^a + b
#  env:  global
#  
#  [1] "quosures"

enquo(s)

  • call from within a function
  • quote what the user passed to an argument as a quosure
f <- function(x) {
  enquo(x)
}
z <- f(x + y)
z
class(z)
#  <quosure>
#  expr: ^x + y
#  env:  global
#  [1] "quosure" "formula"
f <- function(x, y) {
  enquos(x, y)
}
z <- f(x + y, a + b)
z
class(z)
#  <list_of<quosure>>
#  
#  [[1]]
#  <quosure>
#  expr: ^x + y
#  env:  global
#  
#  [[2]]
#  <quosure>
#  expr: ^a + b
#  env:  global
#  
#  [1] "quosures"

sym(s) / ensym(s)

sym(s)

  • convert a string to a symbol
z <- sym("x")
z
class(z)
#  x
#  [1] "name"
  • convert a list of strings to a list of symbols
z <- syms(list("x", "y"))
z
class(z)
#  [[1]]
#  x
#  
#  [[2]]
#  y
#  
#  [1] "list"

ensym(s)

  • call from within a function
  • quote what the user passed to an argument as a symbol
  • accepts strings
