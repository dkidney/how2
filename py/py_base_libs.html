import json

def read_json(path: str):
    try:
        with open(path) as fh:
            obj = json.load(fh)
    except IOError:
        print("error reading json file: \n\t" + path)
    return obj



def write_json(json_dict: dict, filename=None, dir=None):
    if filename is None:
        filename = json_dict["models"][0]["seedName"] + ".json"
    if dir is None:
        path = filename
    else:
        path = dir + "/" + filename
    with open(path, 'w') as fh:
        fh.write(json.dumps(json_dict, indent=2))
    return path
