Rscript -e 'print("hello")'

# scripts
Rscript ~/my_script.R

# options 
Rscript --vanilla ~/my_script.R

# as above, but allows renviron
Rscript --no-save --no-restore --no-site-file --no-init-file ~/my_script.R
