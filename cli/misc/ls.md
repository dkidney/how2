```sh
ls -l # list file info
ls -al # list file info including hidden files
ls -lt # list file info in date order

# wildcards
ls abc* # all files that start with abc
ls ?bc* # all files that start with *bc
ls *.txt # all .txt files 
ls [a-z]*.txt # all .txt files that start with a lower case character
ls {g,d}* # all files that start with a g or d
```
